JavaScript 数组

hdtmk8.github.io @hdtmk8

数组是多个变量值的集合，数组是Array对象的实例，所以可以像对象一样调用方法。

创建数组

使用对象方式创建数组：

console.log(new Array(1,2,3,"hexo","hdtmk8")); //[1,2,3,"hexo","hdtmk8"]

使用字面量创建是推荐的简单作法:

const array = ["hdcms", "houdunren"];

多维数组定义:

const array = [["hexo"], ["hdtmk8"]];
console.log(array[1][0]);

数组是引用类型可以使用const声明并修改它的值 :

const array = ["hexo", "hdtmk8"];
array.push("js");
console.log(array); //["hexo", "hdtmk8","js"]

使用原型的 length属性可以获取数组元素数量 

let hd = ["hexo", "hdtmk8"];
console.log(hd.length); //2

数组可以设置任何值，下面是使用索引添加数组 :

let hd = ["hdtmk8"];
hd[1] = "js";

下面直接设置3号数组，会将1/2索引的数组定义为空值

let hd = ["hdtmk8"];
hd[3] = "js";
console.log(hd.length); //4

声明多个空元素的数组 :

let hd = new Array(3);
console.log(hd.length); //3
console.log(hd); //[empty × 3]

Array.of

使用Array.of 与 new Array 不同是设置一个参数时不会创建空元素数组 :

let hd = Array.of(3);
console.log(hd); //[3]

hd = Array.of(1, 2, 3);
console.log(hd); //[1, 2, 3]

类型检测

检测变量是否为数组类型 :

console.log(Array.isArray([1, "hexo", "hdtmk8"])); //true
console.log(Array.isArray(9)); //false

类型转换

可以将数组转换为字符串也可以将其他类型转换为数组

字符串

大部分数据类型都可以使用.toString()函数转换为字符串：

console.log(([1, 2, 3]).toString()); // 1,2,3

也可以使用函数String转换为字符串：

console.log(String([1, 2, 3])); // 1,2,3

或使用join连接为字符串：

console.log([1,2,3].join("-")) //1-2-3

Array.from

使用Array.from可以将类数组转换为数组，类数组指包含length属性或可迭代的对象。

  • 第一个参数为要转换的数据，第二个参数为类似于map函数的回调方法： 

    let str="hdtmk8";
console.loge(Array.from(str)); //["h","d","t","m","k","8"]

为对象设置length属性后也可以转换为数组，但下标要为数值或数值字符串：

let user={

0:"array",

"1":18,

length:2  //必须设置

};

console.log(Array.from(user)); //["array",18]

展开语法

数组合并

使用展开语法来合并数组相比concat要更简单，使用...可将数组展开为多个值：

let a=[1,2,3];
let b=["a","hdtmk8",...a];
console.logt(b); //["a","hdtmk8",1,2,3] 把a数组的元素全部都添加到b数组元素的后面

函数参数

使用展开语法可以替代arguments来接收任意数量的参数：

function hd(...args){
    console.log(args);
}
hd(1,2,3,"hdtmk8"); //[1,2,3,"hdtmk8"]

也可以用于接收部分参数：

function hd(name,...args){
    console.log(name,args);
}
hd("Alex",1,2,3,"hdtmk8"); //Alex (4)[1,2,3,"hdtmk8"]

解构赋值

解构是一种更简洁的赋值特性，可以理解为分解一个数据的结构。

  • 建议使用let/const声明

基本使用

基本使用语法：

//数组使用
let [name,url]=["hdtmk8","hdtmk8.github.io"];
console.log(name) //hdtmk8

函数使用语法：

function hd(){
    return ["hdtmk8","hexo"];
}
let [a,b]=hd();
console.log(a); //hdtmk8

剩余解构用一个变量来接收剩余参数：

let [a,...b]=["hdtmk8","hexo","alex"];
console.log(b) //["hexo","alex"];

字符串解构：

"use strict";
const [...a]="hdtmk8";
console.log(a); //Array(13)

严格模式，非严格模式可以不使用声明指令，严格模式下必须使用声明 。所以建议使用let等声明：

"use strict"
[web,url]=["hdtmk8","alex"];
console.log(web); //hdtmk8

简洁定义，只赋值部分变量

let [,url]=["hdtmk8","alex";
console.log(url); //hdtmk8

使用展开语法获取多个值：

let [name,...arr]=["hdtmk8","alex","hexo"];
console.log(name,arr); //hdtmk8,["alex","hexo"];

默认值 为变量设置默认值：

let [name,site="hexo"]=["alex"];
console.log(site); //hexo

函数参数 数组参数的使用

function hd([a,b]){

    console.log(a,b);
}
hd(["hddtmk8","alex"]);

合并拆分

join

使用join连接字符串：

let arr=[1,"htdmk8","hexo"];
console.log(arr.join("-")); //1-htdmk8-hexo

split

split方法用于将字符串分割成数组，类似join方法的反函数：

let price="99,97,98";
console.loge(price.split(",")); //["99","97","98"]

concat

concat方法用于连接两个或多个数组，元素是值类型是复制操作，如果是引用类型还是指向同一对象：

let array=["hdtmk8","alex"];
let hd=[1,2];
let cms=[3,4];
console.log(array.concat(hd,cms)); //["hdtmk8","alex",1,2,3,4]

也可以使用扩展语法实现连接：

console.log([...array,...hd,...cms]); //["hdtmk8","alex",1,2,3,4]

copyWithin

使用copyWithin从数组中复制一部分到同数组中的另外位置：

语法说明：

array.copyWithin(target,start,end);

参数说明：

参数 描述
target 必需。复制到指定目标索引位置。
start 可选。元素复制的起始位置。
end 可选。停止复制的索引位置（默认为array.length）。如果是负值，表示倒数。 
const arr=[1,2,3,4];
console.log(arr.copyWithin(2,0,2)); //[1,2,1,2]

查找元素

数组包含多种查找的函数，需要把这些函数掌握清楚，然后根据不同场景选择合适的函数。

indexOf

使用indexOf从前向后查找元素出现的位置，如果找不到返回-1：

let arr=[1,2,3,4,5,6,7];
console.log(arr.indexOf(2)); // 2 从前面查找2出现的位置

indexOf是严格类型约束类似于===：

let arr=[1,2,3,4,5,6,"7"];
console.log(arr.indexOf(7)); //-1 代表没有找到

第二个参数用于指定查找开始的位置：

let arr=[1,2,3,4,5,6,7];
console.log(arr.indexOf(2,3)); //4

IastindexOf

使用lastindexOf从后向前查找元素出现的位置，如果找不到返回-1：

let arr=[5,7,3,7,3];
console.log(arr.lastIndexOf(7)); //3 从后查找7出现的位置

第二个参数用于指定查找开始位置

let arr=[2,3,4,1,2,9];
console.log(arr.lastIndexOf(2,5)); //从第5个元素向前查找
console.loge(arr.lastIndexOf(2,-2)); //从最后一个字符向前查找

includes

使用includes查找字符串返回值是布尔类型更方便判断：

let arr=[7,5,3,1];
console.log(arr.includes(3)); //true

写一个类似includes功能的函数：

function includes(array,item){
    for(const value of array)
        if(item ==== value) return true;
    return false;
}
console.log(includes([1,2,3,4,5],3)); //true

find

find方法找到后会把值返回出来

  • 如果找不到返回值为undefined

返回第一次找到的值，不继续查找

let arr=["hdtmk8","hexo","alex"];
let find=arr.find(function(item){
    return item=="hexo";
});
console.log(find); //hexo

也可以使用find把元素遍历出来：

let arr=["hdtmk8","hexo","alex"];
let find=arr.find(function(item){
    console.log(item)
});
console.log(find); //hexo

使用includes等不能查找引用类型，因为它们的内存地址不是相等的：

const user=[{name:"李四"}，{name:"张飞"}，{name:"alex"}];
const find= user.includes({name:"alex"});
console.log(find); //undefined

find可以方便的查找引用类型：

const user = [{ name: "李四" }, { name: "张三" }, { name: "alex" }];
const find = user.find(user => (user.name = "alex"));
console.log(find); //{name:"alex"}

findIndex

findIndex与find的区别是返回索引 值，参数也是：当前值，索引，操作数组

  • 查找不到时返回-1
let arr=[1,2,3,4,5,6,"7",8];
console.log(arr.findIndex(functio(item){
    return item==8;
})); //7

下面使用自定义函数

let arr=[1,2,3,4,5];
function find(array,callback){
    for(const value of array){
        if(callback(value) === true) return value;
    }
    return undefined;
}
let res=find(arr,fuctdion(item){
    return item == 23;

console.log(res);

数组排序

sort

sort每次使用两个值进行比较Array.sort((a,a)=>a-b)

  • 返回负数a排在b前面，从小到大
  • 返回正数b排在a前面，从大到小
  • 返回0时不动

默认从小到大排序数组元素

let arr=[1,5,4,7];
consosle.log(arr.sort()); //[1,4,5,7]

使用排序函数从大到小排序，参数一与参数二比较，返回正数为降序，负数为升序：

let arr=[1,4,2,9];
console.log(arr.sort(function (v1,v2){
    return v2-v1;
})); //[9,4,2,1]

下面是按课程点击数由高到低排序 ：

let lessons=[
  { title: "媒体查询响应式布局", click: 78 },
  { title: "FLEX 弹性盒模型", click: 12 },
  { title: "MYSQL多表查询随意操作", click: 99 }
];
let sortLessons = lessons.sort((v1, v2) => v2.click - v1.click);
console.log(sortLessons);

排序原理

let arr=[1,5,3,9,7];
function sort(array,callback){
    for(const n in array){
        for(const m in array){
            if(callback(array[n],array[m]<0)){
                let temp=array[n];
                array[n]=array[m];
                array[m]=temp;
            }
        }
    }
    return array;
}
arr=sort(arr,function(a,b){
    return a-b
});
console.table(arr);

反转数组排序

let arr=[1,4,2,9];

循环遍历

根据数组长度结合for循环来遍历数组：

let lessons = [

     {title: 'FLEX 弹性盒模型',category: 'css'},

for (let i = 0; i < lessons.length; i++) {

遍历时的key值为数组的索引：

let lessons = [

    {title: 'MYSQL多表查询随意操作',category: 'mysql'}

forEach使函数作用在每个数组元素上，但是没有返回值，下面盒子是截取标签的一个字符：

扩展方法

every

every用于递归的检测元素，要所有元素操作都要返回真结果才为真。

查看班级中同学的JS成绩是否都及格：

const user = [
  { name: "李四", js: 89 },
  { name: "马六", js: 55 },
  { name: "张三", js: 78 }
];
const resust=user.every(function(user){
    return user.js >=60;
})
console.log(resust);

标题的关键词检查：

let words = ['hdtmk8', 'JS', 'css'];
let title = '我的个人博客地址hdtmk8.github.io';

let state = words.every(function (item, index, array) {
  return title.indexOf(item) >= 0;
});

if (state == false)
    console.log('标题必须包含所有关键词');

some

使用some函数可以递归的检测元素，如果有一个返回true,表达式结果就是真。第一个参数为元素，第二个参数为索引，第三个参数为原数组。

下面是使用some检测规则关键词的示例，如果匹配到一个词就提示违规：

let words = ['hdtmk8', 'JS', 'css'];
let title = '我的个人博客地址hdtmk8.github.io';

let state = words.some(function(item,index,array){
        return title.indexOf(item)>=0;
});
if(state){
    console.log("标题含有违规关键词");
}

filter

使用filter可以过滤数据中元素，下面是获取所有在css栏目的课程：

let lessons = [
  {title: '媒体查询响应式布局',category: 'css'},

let cssLessons=lessons.filter(function(item,index,arr){
    if(item.category.toLowerCase()=='css'){
        return true;
    }

console.log(cssLessons);

过滤元素的方法：

function except(array, excepts) {
  const newArray = [];
  for (const elem of array)
    if (!excepts.includes(elem)) newArray.push(elem);
  return newArray;
}

const array = [1, 2, 3, 4];
console.log(except(array, [2, 3])); //[1,4]

超好用的reduce

使用reduce与reduceRight函数可以迭代数组的所有元素，reduce从前开始，reduceRight从后面开始。

第一个参数是执行函数，第二个参数为初始值

  • 传入第二个参数时将所有元素循环一遍
  • 不传第二个参数时从第二个元素开始循环

函数参数说明如下：

参数 说明
prev 上次调用回调函数返回的结果
cur 当前的元素值
index 当前的索引
array 原数组

统计元素出现的次数：

function countArrayElem(array,elem){
    return array.reduct((total,cur)=>(total+=cur==elem?1:0),);
}
let numbers=[1,2,3,1,5];
console.log(countArrayElem(numbers,1)); //2

取数组中的最大值：

function arrayMax(array){
    return array.reduce(
    (max,elem)=>(max > elem ? max : elem),array[0]);
}
console.log(arrayMax([1,3,2,9]));

取价格最高的商品 :

let cart = [
  { name: "iphone", price: 12000 },
  { name: "imac", price: 25000 },
  { name: "ipad", price: 3600 }
];
function maxPrice(array){
    return array.reduce(
    (goods,elem)=>(goods.price >  elem.price ? goods:elem),
    array[0]);
}
console.log(maxPrice(cart));

计算购物车中商品总价：

let cart = [
  { name: "iphone", price: 12000 },
  { name: "imac", price: 25000 },
  { name: "ipad", price: 3600 }
];
const total=cart.reduce(
    (total,goods)=>total+=goods.price,0);
console.log(total); //40600

获取价格超过1万的商品名称：

let goods = [
  { name: "iphone", price: 12000 },
  { name: "imac", price: 25000 },
  { name: "ipad", price: 3600 }
];

function getNameByPrice(array,price){
    return array.reduce((goods,elem)=>{
        if(elem.price > price){
            goods.push(elem);
        }
        return goods;
    },[]).map(elem=>elem.name);
}
console.table(getNameByPrice(goods,10000));

使用reduce实现数组去重：

let arr=[1,2,6,2,1];
let filterArr=arr.reduce((pre,cur,index,array)=>{
    if(pre.includes(cur)===false){
        pre=[...pre,cur];
    }
