Map与WeakMap

hdtmk8 @hdtmk8

Map

Map是一组键值对的结构，用于解决以往不能用对象做为键的问题。

  • 具有极快的查找速度。
  • 函数，对象，基本类型都可以作为键或值

声明定义

可以接受一个数组作为参数，该数组的成员是一个表示键值对的数组：

let m=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);

console.log(m.get("hdtmk8")); //博客地址
console.log(m); //Map(2) {"hdtmk8" => "博客地址", "alex" => "name"}

使用set方法添加元素，支持链式操作：

let map=new Map();
let obj={
    name:"alex"
};
map.set(obj,"hdtmk8");
map.set(obj, "hdtmk8.github.io").set("name", "make");

console.log(map.entries()); //MapIterator {{…} => "hdtmk8.github.io", "name" => "make"}

对于键是对象的Map， 键保存的是内存地址，值相同但内存地址不同的视为两个键。 

let arr = ["hdtmk8"];
const hd = new Map();
hd.set(arr, "mark");
console.log(hd.get(arr)); //mark
console.log(hd.get(["mark"])); //undefined

获取数量

获取数据数量size：

console.log(map.size);

元素检测

检测元素是否存在has()：

console.log(map.has(hdtmk8));

读取元素

使用get()来读取元素：

let map=new Map();
let obj={
    name:"alex"
};
map.set(obj,"hdtmk8");
console.log(map.get(obj));

删除元素

使用delete()方法删除单个元素：

let map=new Map();
let obj={
    name:"alex"
};
map.set(obj,"hdtmk8");
console.log(map.delete(obj)); //true 成功删除元素返回true
console.log(map.delete(aaa)); //false 没有这个键名

使用clear方法清除Map所有元素：

let map=new Map();
let obj={
    name:"alex"
};
map.set(obj,"hdtmk8");
map.clear();
console.log(map.size);

遍历数据

使用keys()/values()/entries()都可以返回可遍历的迭代对象：

let hd=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);
console.log(hd.keys()); //MapIterator {"hdtmk8", "alex"}
console.log(hd.values()); //MapIterator {"博客地址","name"}
console.log(hd.entries()); //MapIterator{"hdtmk8"=>"博客地址"，"alex"=>"name"}

可以使用keys/values函数遍历键与值：

let hd=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);
for(const key of hd.keys()){
    console.log(key);
}
for(const value of hd.values()){
    console.log(value);
}

使用for/of遍历操作，直接遍历Map等同于使用entries()函数:

let hd=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);
for(const [key,value] of hd){
    console.log(`${key}=>${value}`);
}

使用fonEach遍历操作：

let hd=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);
hd.forEach(function(key,value){
    console.log(`${key}=>${value}`);
})

数组转换

可以使用展开语法或Array.form静态方法将Map类型转为数组，这样就可以使用数组处理函数了：

let hd=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);
console.log(...hd); //(2) ["hdtmk8","博客地址"](2)["alex","name"]
console.log(...hd.entries()); //(2) ["hdtmk8","博客地址"](2)["alex","name"]
console.log(...hd.values()); //博客地址 alex
console.log(...hd.keys()); //hdtmk8 name

检索包含博客地址的值组成新Map：

let hd=new Map([
    ["hdtmk8","博客地址"],
    ["alex","name"]
]);

let newArr=[...hd].filter(function(item){
    return item[1].includes("博客地址");
});

hd=new Map(newArr);
console.log(...hd.keys());

WeakMap

WeakMap 对象是一组键/值对的集

  • 键名必须是对象(引用类型)

  • WeaMap对键名是弱引用的，键值是正常引用

  • 垃圾回收不考虑WeaMap的键名，不会改变引用计数器，键在其他地方不被引用时即删除

  • 因为WeakMap 是弱引用，由于其他地方操作成员可能会不存在，所以不可以进行forEach( )遍历等操作

  • 也是因为弱引用，WeaMap 结构没有keys( )，values( )，entries( )等方法和 size 属性

  • 当键的外部引用删除时，系统自动删除数据时使用 WeakMap

声明定义

let hd=new WeakMap();
let arr=["bogl"]
hd.set(arr,"hdtmk8");
console.log(hd.has(arr)); //true

基本操作

下面是WeakMap的常用指令：

const hd=new WeakMap();
const arr=["bogl"];

//添加操作
hd.set(arr,"hdtmk8");
//删除操作
hd.delete(arr);
//检索判断
hd.has(arr)

垃圾回收

WeakMap的键名对象不会增加引用计数器，如果一个对象不被引用了会自动删除。

  • 下面当hd删除时内存即清除，因为WeakMap的弱引用不会产生引用计数，当引用计数为0时，系统会自动回收这个内存地址。

  • 当垃圾回收时因为对象被删除，这里WeakMap也就没有记录了。

    let map=new WeakMap();
let hd={};
map.set(hd,"hdtmk8");
hd=null; //把hd删除
console.log(map);

setTimeout(()=>{
    console.log(map)
},1000);

<script>
  class Lesson {
    constructor() {
      this.lis = document.querySelectorAll("ul>li");
      this.countELem = document.getElementById("count");
      this.listElem = document.getElementById("lists");
      this.map = new WeakMap();
    }
    run() {
      this.lis.forEach(item => {
        item.querySelector("a").addEventListener("click", event => {
          const elem = event.target;
          const state = elem.getAttribute("select");
          if (state) {
            elem.removeAttribute("select");
            this.map.delete(elem.parentElement);
            elem.innerHTML = "+";
            elem.style.backgroundColor = "green";
          } else {
            elem.setAttribute("select", true);
            this.map.set(elem.parentElement, true);
            elem.innerHTML = "-";
            elem.style.backgroundColor = "red";
          }
          this.render();
        });
      });
    }
    count() {
      return [...this.lis].reduce((count, item) => {
        return (count += this.map.has(item) ? 1 : 0);
      }, 0);
    }
    lists() {
      return [...this.lis]
        .filter(item => {
          return this.map.has(item);
        })
        .map(item => {
          return `<span>${item.querySelector("span").innerHTML}</span>`;
        });
    }
    render() {
      this.countELem.innerHTML = `共选了${this.count()}课`;
      this.listElem.innerHTML = this.lists().join("");
    }
  }
  new Lesson().run();
</script>
